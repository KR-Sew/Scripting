Invoke-Expression @'
$ErrorActionPreference = 'Stop'

function Enable-Tls12 {
    Write-Host 'Checking TLS 1.2 support...' -ForegroundColor Cyan

    try {
        $tls12 = [Net.SecurityProtocolType]::Tls12
        $currentProtocols = [Net.ServicePointManager]::SecurityProtocol

        if (($currentProtocols -band $tls12) -eq $tls12) {
            Write-Host "TLS 1.2 is already enabled: $currentProtocols" -ForegroundColor Green
            return
        }

        Write-Host "TLS 1.2 is not currently enabled: $currentProtocols" -ForegroundColor Yellow
        Write-Host 'Enabling TLS 1.2 for the current PowerShell process...' -ForegroundColor Cyan

        [Net.ServicePointManager]::SecurityProtocol = $currentProtocols -bor $tls12

        $updatedProtocols = [Net.ServicePointManager]::SecurityProtocol

        if (($updatedProtocols -band $tls12) -ne $tls12) {
            throw 'TLS 1.2 could not be enabled.'
        }

        Write-Host "TLS 1.2 enabled successfully: $updatedProtocols" -ForegroundColor Green
    }
    catch {
        throw "Failed to configure TLS 1.2: $($_.Exception.Message)"
    }
}

function Test-IsAdministrator {
    $identity = [Security.Principal.WindowsIdentity]::GetCurrent()
    $principal = [Security.Principal.WindowsPrincipal]::new($identity)

    return $principal.IsInRole(
        [Security.Principal.WindowsBuiltInRole]::Administrator
    )
}

function Remove-TemporaryInstaller {
    param(
        [Parameter(Mandatory)]
        [string]$Path
    )

    if (Test-Path -LiteralPath $Path) {
        try {
            Remove-Item -LiteralPath $Path -Force -ErrorAction Stop
        }
        catch {
            Write-Warning "Could not remove temporary installer '$Path': $($_.Exception.Message)"
        }
    }
}

if (-not (Test-IsAdministrator)) {
    Write-Error 'This script must be run from an elevated PowerShell session.'
    return
}

Enable-Tls12

Write-Host 'Checking current PowerShell version...' -ForegroundColor Cyan

$currentVersion = [version]$PSVersionTable.PSVersion
Write-Host "Current version: $currentVersion"

Write-Host 'Fetching latest PowerShell release from GitHub...' -ForegroundColor Cyan

try {
    $release = Invoke-RestMethod `
        -Uri 'https://api.github.com/repos/PowerShell/PowerShell/releases/latest' `
        -UseBasicParsing `
        -Headers @{
            Accept       = 'application/vnd.github+json'
            'User-Agent' = 'PowerShell-Update-Script'
        }
}
catch {
    Write-Error "Failed to fetch the latest release: $($_.Exception.Message)"
    return
}

try {
    $latestVersionText = [string]$release.tag_name

    if ([string]::IsNullOrWhiteSpace($latestVersionText)) {
        throw 'GitHub returned an empty release version.'
    }

    $latestVersion = [version]$latestVersionText.TrimStart('v')
}
catch {
    Write-Error "Could not determine the latest PowerShell version: $($_.Exception.Message)"
    return
}

Write-Host "Latest version available: $latestVersion"

if ($latestVersion -le $currentVersion) {
    Write-Host 'You are already using the latest PowerShell version.' -ForegroundColor Green
    return
}

$msiAsset = $release.assets |
    Where-Object {
        $_.name -match '^PowerShell-[\d.]+-win-x64\.msi$'
    } |
    Select-Object -First 1

if (-not $msiAsset) {
    Write-Error 'Could not find the Windows x64 MSI installer in the latest release.'
    return
}

$msiUrl = [string]$msiAsset.browser_download_url
$msiFile = Join-Path `
    -Path $env:TEMP `
    -ChildPath "PowerShell-$latestVersion-win-x64.msi"

Write-Host "Downloading installer from:`n$msiUrl" -ForegroundColor Cyan
Write-Host "Destination: $msiFile"

Remove-TemporaryInstaller -Path $msiFile

try {
    Invoke-WebRequest `
        -Uri $msiUrl `
        -OutFile $msiFile `
        -UseBasicParsing `
        -Headers @{
            Accept       = 'application/octet-stream'
            'User-Agent' = 'PowerShell-Update-Script'
        }
}
catch {
    Remove-TemporaryInstaller -Path $msiFile
    Write-Error "Failed to download the MSI installer: $($_.Exception.Message)"
    return
}

if (-not (Test-Path -LiteralPath $msiFile)) {
    Write-Error "The MSI installer was not created: $msiFile"
    return
}

$downloadedFile = Get-Item -LiteralPath $msiFile

if ($downloadedFile.Length -eq 0) {
    Remove-TemporaryInstaller -Path $msiFile
    Write-Error 'The downloaded MSI installer is empty.'
    return
}

Write-Host "Downloaded $([math]::Round($downloadedFile.Length / 1MB, 2)) MB." `
    -ForegroundColor Green

Write-Host 'Scheduling PowerShell update task...' -ForegroundColor Cyan

$taskName = 'UpdatePowerShell'

# Use an argument list rather than cmd.exe to avoid unnecessary quoting layers.
$msiArguments = "/i `"$msiFile`" /quiet /norestart"

try {
    $existingTask = Get-ScheduledTask `
        -TaskName $taskName `
        -ErrorAction SilentlyContinue

    if ($existingTask) {
        Write-Host "Removing existing scheduled task '$taskName'..." -ForegroundColor Yellow

        Unregister-ScheduledTask `
            -TaskName $taskName `
            -Confirm:$false `
            -ErrorAction Stop
    }

    $action = New-ScheduledTaskAction `
        -Execute "$env:SystemRoot\System32\msiexec.exe" `
        -Argument $msiArguments

    $trigger = New-ScheduledTaskTrigger `
        -Once `
        -At ((Get-Date).AddMinutes(1))

    $principal = New-ScheduledTaskPrincipal `
        -UserId 'SYSTEM' `
        -LogonType ServiceAccount `
        -RunLevel Highest

    $settings = New-ScheduledTaskSettingsSet `
        -ExecutionTimeLimit (New-TimeSpan -Minutes 30) `
        -StartWhenAvailable

    Register-ScheduledTask `
        -TaskName $taskName `
        -Action $action `
        -Trigger $trigger `
        -Principal $principal `
        -Settings $settings `
        -Description "Install PowerShell $latestVersion" |
        Out-Null
}
catch {
    Write-Error "Failed to schedule the update task: $($_.Exception.Message)"
    return
}

Write-Host ''
Write-Host 'Update task created successfully.' -ForegroundColor Green
Write-Host "Installer: $msiFile"
Write-Host "Task name: $taskName"
Write-Host ''
Write-Host 'Please disconnect the current PowerShell session now.' -ForegroundColor Yellow
Write-Host 'The update is scheduled to start in approximately one minute.' -ForegroundColor Yellow
'@
